---
apiVersion: v1
kind: Service
metadata:
  name: [[${projectName}]]-k8s
spec:
  type: LoadBalancer
  selector:
    # This should match the app name from deployment section
    app: [[${projectName}]]
  ports:
      # By default and for convenience, the `targetPort` is set to the same value as the `port` field.
    - port: 80
      targetPort: [[${appPort}]]
      # Optional field
      # By default and for convenience, the Kubernetes control plane will allocate a port from a range (default: 30000-32767)
      # nodePort: 30080
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: [[${projectName}]]-deployment
  labels:
    app: [[${projectName}]]
spec:
  replicas: 2
  selector:
    matchLabels:
      app: [[${projectName}]]
  template:
    metadata:
      labels:
        app: [[${projectName}]]
    spec:
      containers:
      - name: [[${projectName}]]
        image: selvendiranj/[[${projectName}]]
        ports:
        - containerPort: [[${appPort}]]
